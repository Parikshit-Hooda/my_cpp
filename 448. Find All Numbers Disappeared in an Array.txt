class Solution {
public:
    vector<int> findDisappearedNumbers(vector<int>& nums) {
        vector<bool> seen(nums.size(), false);

        for(auto ele: nums) {
            seen[ele-1] = true;
        }
        vector<int> res;
        for(int i = 0; i < seen.size(); i++) {
            if(!seen[i]) res.push_back(i+1);
        }
        return res;
    }
};
